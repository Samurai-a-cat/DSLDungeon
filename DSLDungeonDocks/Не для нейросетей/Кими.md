Твоя задача - подключить мой проект через команду:

```python
import subprocess
import os

# Try cloning with a longer timeout approach
process = subprocess.Popen(
    ['bash', '-c', 'GIT_SSL_NO_VERIFY=1 git clone --depth 1 https://github.com/Samurai-a-cat/DSLDungeon.git /mnt/agents/output/DSLDungeon'],
    stdout=subprocess.PIPE, stderr=subprocess.PIPE, text=True
)

try:
    stdout, stderr = process.communicate(timeout=60)
    print("STDOUT:", stdout)
    print("STDERR:", stderr)
    print("Return code:", process.returncode)
except subprocess.TimeoutExpired:
    process.kill()
    print("Process timed out, trying to check partial clone...")
```
